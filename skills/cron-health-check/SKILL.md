# Cron Health Check
\nexport default function CronHealthCheck() {\n  const status = {\n    jobs: [] as any[],\n    alerts: [] as string[],\n  };\n\n  // Check if cron is responding\n  const cronOk = await invoke(\"openclaw cron list\").catch(() => null);\n  \n  if (!cronOk || !cronOk.jobs) {\n    status.alerts.push(\"Cron service not responding\");\n    return status;\n  }\n\n  status.jobs = cronOk.jobs;\n\n  // Check for failed jobs\n  const failed = cronOk.jobs.filter(\n    (j: any) => j.state?.lastRunStatus === \"error\"\n  );\n  \n  if (failed.length > 0) {\n    status.alerts.push(`${failed.length} jobs failed recently`);\n  }\n\n  // Check for disabled jobs that should be enabled\n  const disabled = cronOk.jobs.filter((j: any) => !j.enabled);\n  \n  // Check for jobs that haven't run in expected timeframe\n  const now = Date.now();\n  const stale = cronOk.jobs.filter((j: any) => {\n    if (!j.enabled) return false;\n    const lastRun = j.state?.lastRunAtMs || 0;\n    return now - lastRun > 7 * 24 * 60 * 60 * 1000; // 7 days\n  });\n\n  return {\n    ...status,\n    failedJobs: failed,\n    disabledJobs: disabled,\n    staleJobs: stale,\n  };\n}\n

All enabled jobs have run recently\n// 2. No jobs stuck in 'error' state\n// 3. No duplicate jobs\n// 4. Next run times are reasonable\n
